# first we need to make sure we need to provide as our minimum version
cmake_minimum_required(VERSION 3.27.7)

# define project we have
project(LightYearsNxt
		VERSION 1.0.0  
		# last 0 is bugs fixes
		# second 0 is same version but new features
		# first number 1 means is version changes.
		LANGUAGES C CXX
)

# need to setup C++ version
set(CMAKE_CXX_STANDARD 14)
# to enforce this
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS       OFF)




# Fetching the SFML library to build our framework
# to do that we need to include function called FetchContent
include(FetchContent)

# now we will set variable for SFML
set(SFML_LIB_NAME SFML)

FetchContent_Declare(
	#variable name
	${SFML_LIB_NAME}
	# location
	GIT_REPOSITORY https://github.com/SFML/SFML
	# to fetch specific version
	GIT_TAG 2.5.1
	# to avoid the shallow copy and ask actually download referenced library if in anycase SFML is using any other libraries 
	GIT_SHALLOW FALSE
)

FetchContent_MakeAvailable(${SFML_LIB_NAME})

# setting the variables for valu LightYearsGame
set(LIGHT_YEARS_ENGINE_TARGET_NAME LightYearsEngine)
set(LIGHT_YEARS_GAME_TARGET_NAME LightYearsGame)
# adding subdirectory
add_subdirectory(LightYearsNxtEngine)
add_subdirectory(LightYearsNxtGame)
