# first we need to make sure we need to provide as our minimum version
cmake_minimum_required(VERSION 3.27.7)

# define project we have
project(LightYearsNxt
		VERSION 1.0.0  
		# last 0 is bugs fixes
		# second 0 is same version but new features
		# first number 1 means is version changes.
		LANGUAGES C CXX
)

# need to setup C++ version
set(CMAKE_CXX_STANDARD 14)
# to enforce this
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS       OFF)

# setting the variables for valu LightYearsGame
set(LIGHT_YEARS_GAME_TARGET_NAME LightYearsGame)

# adding subdirectory
add_subdirectory(LightYearsNxt)
